import React from "react";
import DefaultLayoutNarrow from "@/components/default-layout-narrow";
import { Box, Text } from "grommet";

const reports = [
  { year: "2022", link: "/report/2022-report" },
  { year: "2021", link: "/report/2021-report" },
  { year: "2020", link: "/report/2020-report" },
  { year: "2019", link: "/report/2019-report" },
];

export default function Layout({ children }) {
  return (
    <DefaultLayoutNarrow>
      {children}
      <Box pad="small">
        {/* Render the list */}
        {reports.map((report) => (
          <Box
            key={report.year}
            direction="row"
            justify="between"
            align="center"
            pad={{ vertical: "small", horizontal: "medium" }}
            margin={{ bottom: "small" }}
            border={{ color: "#eaeaea", size: "small", radius: "8px" }}
            background="white"
            hoverIndicator={{ background: "#f5f5f5", transform: "scale(1.02)" }}
            onClick={() => (window.location.href = report.link)} // Navigate to the link on click
            onMouseOut={(e) => {
              e.currentTarget.style.transform = "scale(1)"; // Reset scale
            }}
          >
            {/* Year on the left */}
            <Text weight="bold" size="16px" color="#333">
              {report.year}
            </Text>

            {/* Title on the right */}
            <Text size="16px" color="#007acc" style={{ textDecoration: "none" }}>
              Annual Report
            </Text>
          </Box>
        ))}
      </Box>
    </DefaultLayoutNarrow>
  );
}
